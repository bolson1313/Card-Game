/**
 * Callback wywoływany przy zmianie widoczności elementów w obszarze widoczności przeglądarki.
 * @callback IntersectionObserverCallback
 * @param {IntersectionObserverEntry[]} entries - Tablica obiektów reprezentujących elementy w obszarze widoczności.
 * @returns {void}
 */

/**
 * Obiekt IntersectionObserver sprawdzający, czy elementy są w obszarze widoczności.
 * @type {IntersectionObserver}
 */
const isVisible = new IntersectionObserver((entries) => {
    /**
     * Iteruje przez każdy wpis w tablicy entries.
     * @param {IntersectionObserverEntry} entry - Obiekt reprezentujący pojedynczy wpis w obszarze widoczności.
     * @returns {void}
     */
    entries.forEach((entry) => {
        console.log(entry);

        /**
         * Sprawdza, czy element jest widoczny w obszarze przeglądarki.
         * @type {boolean}
         */
        if (entry.isIntersecting) {
            console.log("show");

            /**
             * Dodaje klasę 'show' do elementu, gdy jest widoczny.
             * @type {DOMTokenList}
             */
            entry.target.classList.add('show');
        } else {
            console.log("hide");

            /**
             * Usuwa klasę 'show' z elementu, gdy nie jest widoczny.
             * @type {DOMTokenList}
             */
            entry.target.classList.remove('show');
        }
    });
});

/**
 * Iteruje przez każdy element kolekcji 'hiddenSection' i obserwuje ich widoczność.
 * @param {Element} element - Element HTML o klasie 'hidden'.
 * @returns {void}
 */
hiddenSection.forEach((element) => isVisible.observe(element));
